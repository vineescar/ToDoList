sudo dnf update -y

# 2️⃣ Install Docker
sudo dnf install docker -y

# 3️⃣ Start Docker service
sudo systemctl start docker
sudo systemctl enable docker  # Ensures Docker starts on reboot

# 4️⃣ Add EC2 user to Docker group (so you don't need sudo for Docker commands)
sudo usermod -aG docker ec2-user

# 5️⃣ Restart instance to apply changes
sudo reboot
